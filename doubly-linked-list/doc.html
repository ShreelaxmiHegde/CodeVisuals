Doubly Linked List

Bidirectional Linked List

  • Can traverse both forward and backward.
  • Each node has two pointers: next and prev.
  • Each node stores: data, next, and prev.

Implementation

        class Node {
        public:
            int data;
            Node* next;
            Node* prev;

            Node(int val) {
                data = val;
                next = prev = NULL;
            }
        };

        class DoublyList {
        public:
            Node* head;
            Node* tail;

            DoublyList() {
                head = tail = NULL;
            }

            // Insert at the beginning
            void push_front(int val) {
                Node* newNode = new Node(val);

                if (head == NULL) { // Empty list
                    head = tail = newNode;
                } else { // Non-empty list
                    newNode->next = head;
                    head->prev = newNode;
                    head = newNode;
                }
            }

            // Remove from the beginning
            void pop_front() {
                if (head == NULL) return; // Empty list

                Node* temp = head;
                head = head->next;

                if (head != NULL) {
                    head->prev = NULL;
                } else {
                    tail = NULL; // List became empty
                }

                temp->next = NULL;
                delete temp;
            }

            // Similarly, push_back and pop_back can be implemented
        };
